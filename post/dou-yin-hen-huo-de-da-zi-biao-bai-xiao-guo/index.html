抖音很火的打字表白效果

2019-05-09

代码：

<!DOCTYPE html>
<html>

<head>
    <meta charset="UTF-8">
    <title>Document</title>
</head>

<body>
    <div id="divTyping"></div>
    <script>
        var str = '<big>不是在最美的时光遇到你，<br />而是遇到你之后都是最美的时光！</big>';
        var i = 0;
        function typing() {
            var divTyping = document.getElementById('divTyping');
            if (i < str.length) {
                divTyping.innerHTML = str.slice(0, i++) + '_';
                setTimeout('typing()', 200);
            }
            else {
                divTyping.innerHTML = str;
            }
        }
        typing();
    </script>
</body>

</html>
